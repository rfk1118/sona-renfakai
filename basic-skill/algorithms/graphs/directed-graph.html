func (g *Graph) noSuccessors() int {
  var isEdge bool
  vertex := g.nVertex
  for row := 0; row < vertex; row++ {
    isEdge = false
    for col := 0; col < vertex; col++ {
      if g.adjMat[row][col] > 0 {
        isEdge = true
        break
      }
    }
    if !isEdge {
      return row
    }
  }
  return -1
}
